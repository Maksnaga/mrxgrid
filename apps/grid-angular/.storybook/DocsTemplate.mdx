import { Meta, Source, useOf, Title, Subtitle, Description, Primary, Controls, Stories, Heading } from '@storybook/addon-docs/blocks';
import { background } from 'storybook/theming';

export const LocalImport = ({ of }) => {
  const resolvedOf = useOf(of || 'story', ['story', 'meta']);
  const story = resolvedOf.story;
  const componentExport = story?.component ?? resolvedOf.meta?.component;
  const rawComponentName =
    componentExport?.name ??
    componentExport?.__name ??
    componentExport?.displayName ??
    story?.name ??
    'Component';
  const componentName = rawComponentName.replace(/^_+/, '');

  const importCode = `import { ${componentName} } from '@mozaic-ds/angular';
 `;
  if (resolvedOf.type === 'story') {
    if (componentName.includes('Release32')) {
      return (

        <>
          <div>The icons are available as a named export from the <code>@mozaic-ds/icons-angular</code> package.<br />
            So first install the package :</div>
          <Source dark language='bash' code={`npm install @mozaic-ds/icons-angular`} />
          <div>Use the following import to include an icon in your project.</div>
          <Source dark language='tsx' code={`
import { Search24, Close16 } from '@mozaic-ds/icons-angular';
<Search24/>
<Close16/>
          `} />
        </>
      );
    } else {
      return (
        <>
          <div>Use the following import to include the component in your project.<br />
            The component is available as a named export from the `@mozaic-ds/angular` package.</div>
          <Source dark code={`${importCode}`} />
        </>
      );
    }

  }
  return null;
};

<Meta isTemplate />

<Title />
<Subtitle />
<Description />

<Primary />

<Controls />

export const ConditionalImport = ({ of }) => {
  const resolvedOf = useOf(of || 'story', ['story', 'meta']);
  const parameters = resolvedOf.story?.parameters ?? resolvedOf.meta?.parameters ?? {};

  if (parameters?.hideImport) {
    return null;
  }
  return (
    <>
      <Heading>Import</Heading>
      <LocalImport of={of} />
    </>
  );
};

<ConditionalImport />
